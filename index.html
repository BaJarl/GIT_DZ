Посилання на галерею

git --version - перегляд версії

git config user.name - перевірити ім'я в конфіг-файлі

git config user.email - перевірити пошту в конфіг-файлі

git config --global user.email - встановити пошту в конфіг-файлі

git config --global user.name - встановити ім'я в конфіг-файлі

git remote add origin 'url' - вказати адрес гіт

git push origin master - створити гілку 'master'

git branch -v - перегляд активної гілки

git checkout -b 'name' - перейти на іншу гілку

git init - створення репозиторію

git add - додати файл під версіонний контроль

git commit -m 'name' - підтвердити зміни

git status - переглянути статус файлів

git log - переглянути історію змін
